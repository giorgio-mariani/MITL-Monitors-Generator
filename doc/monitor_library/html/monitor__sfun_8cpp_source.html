31 /* Error handling
32  * --------------
33  *
34  * You should use the following technique to report errors encountered within
35  * an S-function:
36  *
37  * ssSetErrorStatus(S,"error encountered due to ...");
38  * return;
39  *
40  * Note that the 2nd argument to ssSetErrorStatus must be persistent memory.
41  * It cannot be a local variable in your procedure. For example the following
42  * will cause unpredictable errors:
43  *
44  * mdlOutputs()
45  * {
46  * char msg[256]; {ILLEGAL: to fix use "static char msg[256];"}
47  * sprintf(msg,"Error due to %s", string);
48  * ssSetErrorStatus(S,msg);
49  * return;
50  * }
51  *
52  * The ssSetErrorStatus error handling approach is the suggested alternative
53  * to using the mexErrMsgTxt function. MexErrMsgTxt uses "exception handling"
54  * to immediately terminate S-function execution and return control to
55  * Simulink. In order to support exception handling inside of S-functions,
56  * Simulink must setup exception handlers prior to each S-function invocation.
57  * This introduces overhead into simulation.
58  *
59  * If you do not call mexErrMsgTxt or any other routines that cause exceptions,
60  * then you should use SS_OPTION_EXCEPTION_FREE_CODE S-function option. This
61  * is done by issuing the following command in the mdlInitializeSizes function:
62  *
63  * ssSetOptions(S, SS_OPTION_EXCEPTION_FREE_CODE);
64  *
65  * Setting this option, will increase the performance of your S-function by
66  * allowing Simulink to bypass the exception handling setup that is usually
67  * performed prior to each S-function invocation. Extreme care must be taken
68  * to verify that your code is exception free when using the
69  * SS_OPTION_EXCEPTION_FREE_CODE option. If your S-function generates
70  * an exception when this option is set, unpredictable results will occur.
71  *
72  * Exception free code refers to code which never "long jumps". Your S-function
73  * is not exception free if it contains any routine which when called has
74  * the potential of long jumping. For example mexErrMsgTxt throws an exception
75  * (i.e. long jumps) when called, thus ending execution of your S-function.
76  * Use of mxCalloc may cause unpredictable problems in the event of a memory
77  * allocation error since mxCalloc will long jump. If memory allocation is
78  * needed, you should use the stdlib.h calloc routine directly and perform
79  * your own error handling.
80  *
81  * All mex* routines have the potential of long jumping (i.e. throwing an
82  * exception). In addition several mx* routines have the potential of
83  * long jumping. To avoid any difficulties, only the routines which get
84  * a pointer or determine the size of parameters should be used. For example
85  * the following will never throw an exception: mxGetPr, mxGetData,
86  * mxGetNumberOfDimensions, mxGetM, mxGetN, mxGetNumberOfElements.
87  *
88  * If all of your "run-time" methods within your S-function are exception
89  * free, then you can use the option:
90  * ssSetOptions(S, SS_OPTION_RUNTIME_EXCEPTION_FREE_CODE);
91  * The other methods in your S-function need not be exception free. The
92  * run-time methods include any of the following:
93  * mdlGetTimeOfNextVarHit, mdlOutputs, mdlUpdate, and mdlDerivatives
94  *
95  * Warnings & Printf's
96  * -------------------
97  * You can use ssWarning(S,msg) to display a warning.
98  * - When the S-function is compiled via mex for use with Simulink,
99  * ssWarning equates to mexWarnMsgTxt.
100  * - When the S-function is used with Real-Time Workshop,
101  * ssWarning(S,msg) equates to
102  * printf("Warning: in block '%s', '%s'\n", ssGetPath(S),msg);
103  * if the target has stdio facilities, otherwise it becomes a comment and
104  * is disabled.
105  *
106  * You can use ssPrintf(fmt, ...) to print a message.
107  * - When the S-function is compiled via mex for use with Simulink,
108  * ssPrintf equates to mexPrintf.
109  * - When the S-function is used with Real-Time Workshop,
110  * ssPrintf equates to printf, if the target has stdio facilities,
111  * otherwise it becomes a call to a empty function (rtPrintfNoOp).
112  * - In the case of Real-Time Workshop which may or may not have stdio
113  * facilities, to generate the most efficient code use:
114  * #if defined(SS_STDIO_AVAILABLE)
115  * ssPrintf("my message ...");
116  * #endif
117  * - You can also use this technique to do other standard I/O related items,
118  * such as:
119  * #if defined(SS_STDIO_AVAILABLE)
120  * if ((fp=fopen(file,"w")) == NULL) {
121  * ssSetErrorStatus(S,"open failed");
122  * return;
123  * }
124  * ...
125  * #endif
126  */

366  /*
367  * If the the number of expected input parameters is not equal
368  * to the number of parameters entered in the dialog box return.
369  * Simulink will generate an error indicating that there is a

372  return;

388  * The following options summarize different ways for setting the input

390  *
391  * (1) If the input port dimensions are unknown, use
392  * ssSetInputPortDimensionInfo(S, inputPortIdx, DYNAMIC_DIMENSION))

403  *
404  * (4) Otherwise use:
405  * ssSetInputPortDimensionInfo(S, inputPortIdx, dimsInfo)

407  * dimensions. dimsInfo is a structure containing width, number of
408  * dimensions, and dimensions of the port.

410  if(!ssSetInputPortVectorDimension(S, inputPortIdx, DYNAMICALLY_SIZED)) return;

522 /* Function: mdlOutputs =======================================================
523  * Abstract:
524  * In this function, you compute the outputs of your S-function
525  * block. Generally outputs are placed in the output vector(s),
526  * ssGetOutputPortSignal.

528 static void mdlOutputs(SimStruct *S, int_T tid)
529 {
530  FormulaValidator *&formulaPtr = getFormulaPtr(S); /* get formula pointer*/
531  vector<boolean_T> *&vectorPtr = getVectorPtr(S); /* get predicate vector pointer*/
532 
533  InputPtrsType inputs = getInputPortSig(S); /* input values*/
534 
535  /* Updating the vector -------------------------------------------------------*/
536  vector<boolean_T>::iterator it = vectorPtr->begin(),end = vectorPtr->end();
537  for(int_T i=0; i<getInputPortWidth(S); i++, it++)
538  {
539  mxAssert(it != end,"Error in mdlOutputs: Input width different than vector size");
540  boolean_T b = *(static_cast<const boolean_T* const>(inputs[i]));
541  *it = b;
542  }
543 
544  /* Updating the formula validator---------------------------------------------*/
545  try{
546  if(formulaPtr->hasEmptyTrace())
547  formulaPtr->initialConditions(ssGetT(S),*vectorPtr);
548  else
549  formulaPtr->extendTrace(ssGetT(S), *vectorPtr);
550 
551  /* Updating the output---------------------------------------------*/
552  boolean_T &y = getOutputPortSig(S);
553  y = !(formulaPtr->alwaysTrue());
554  }
555  catch(exception &e)
556  {
557  mexErrMsgTxt(e.what());
558  }
559 }
